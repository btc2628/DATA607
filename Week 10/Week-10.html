tidy_books <- austen_books() %>%
  group_by(book) %>%
  mutate(
    linenumber = row_number(),
    chapter = cumsum(str_detect(text, 
                                regex("^chapter [\\divxlc]", 
                                      ignore_case = TRUE)))) %>%
  ungroup() %>%
  unnest_tokens(word, text)
books_by_chapter <- tidy_books %>%
  group_by(book, chapter) %>%
  summarise(text = paste(word, collapse = " ")) %>%
  ungroup()

sentiments <- sentimentr::sentiment_by(books_by_chapter$text)
## Warning: Each time `sentiment_by` is run it has to do sentence boundary disambiguation when a
## raw `character` vector is passed to `text.var`. This may be costly of time and
## memory.  It is highly recommended that the user first runs the raw `character`
## vector through the `get_sentences` function.
sentiments <- sentiments %>%
  mutate(sentiment = ifelse(ave_sentiment > 0, "Positive", "Negative"))
books_by_chapter$element_id <- 1:nrow(books_by_chapter)
sentiment_by_chapter <- merge(books_by_chapter, sentiments, by = "element_id")
ggplot(sentiment_by_chapter, aes(x=chapter, y=ave_sentiment, group=book, fill=book)) + 
  geom_col(show.legend = FALSE) +
  facet_wrap(~book, ncol = 2, scales = 'free_x') + 
  labs(x='Chapter', y='Average Sentiment')

Conclusion

From both sentiment analysis results we can see that the majority of Jane Austins work can be interpreted as having a positive sentiment, with Persuasion not having a single negative chapter, and all of the others having very few wqhen compared to the number of postive chapters. This aligns with what we found when looking at the books by work, with the vast majority of words in each book having a positive sentiment.
